# AutoUpdateModule
自动更新组件
## 添加权限
    <!-- 在SDCard中写入权限 -->
    <uses-permission android:name="android.permission.WRITE_EXTERNAL_STORAGE"/>
    <!-- 在SDCard中创建与删除文件权限 -->
    <uses-permission android:name="android.permission.MOUNT_UNMOUNT_FILESYSTEMS"/>
    <!--网络权限-->
    <uses-permission android:name="android.permission.INTERNET"/>
